package com.example.appjava;

import android.content.Context;
import android.view.*;
import android.widget.*;

public class AdaptadorBebidas extends BaseAdapter {

    Context context;
    String[] nombres;
    int[] imagenes;

    public AdaptadorBebidas(Context context, String[] nombres, int[] imagenes) {
        this.context = context;
        this.nombres = nombres;
        this.imagenes = imagenes;
    }

    public int getCount() {
        return nombres.length;
    }

    public Object getItem(int i) {
        return nombres[i];
    }

    public long getItemId(int i) {
        return i;
    }

    public View getView(int i, View view, ViewGroup parent) {

        LayoutInflater inflater = LayoutInflater.from(context);
        View vista = inflater.inflate(R.layout.grid_item, null);

        ImageView img = vista.findViewById(R.id.img);
        TextView txt = vista.findViewById(R.id.txt);

        img.setImageResource(imagenes[i]);
        txt.setText(nombres[i]);

        return vista;
    }
}

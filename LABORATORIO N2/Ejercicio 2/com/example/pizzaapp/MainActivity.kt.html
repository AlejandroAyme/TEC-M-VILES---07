package com.example.pizzaapp

import android.os.Bundle
import android.widget.GridView
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    private lateinit var gridView: GridView

    private val nombres = arrayOf(
        "Hawaiana", "Pepperoni", "Suprema", "Vegetariana"
    )

    private val imagenes = arrayOf(
        R.drawable.pizza1,
        R.drawable.pizza2,
        R.drawable.pizza3,
        R.drawable.pizza4
    )

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        gridView = findViewById(R.id.gridView)

        val adaptador = AdaptadorPizza(this, nombres, imagenes)
        gridView.adapter = adaptador

        gridView.setOnItemClickListener { _, _, position, _ ->
            Toast.makeText(this, "Elegiste: " + nombres[position] + " 🍕", Toast.LENGTH_SHORT).show()
        }
    }
}
